#include <iostream>
using namespace std;

int main() {
    // Declare variables
    int myInt = 25;         // Integer variable
    float myFloat = 3.14;   // Floating-point variable
    char myChar = 'A';      // Character variable

    // Print the values
    cout << "Integer value: " << myInt << endl;
    cout << "Float value: " << myFloat << endl;
    cout << "Character value: " << myChar << endl;

    return 0;
}
